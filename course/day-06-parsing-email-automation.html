1) Email-отправка результатов

services/email_report.py
import smtplib
from email.mime.text import MIMEText


def send_report_email(sender: str, password: str, receiver: str, subject: str, body: str):
    # Формируем email
    msg = MIMEText(body, "plain", "utf-8")
    msg["Subject"] = subject
    msg["From"] = sender
    msg["To"] = receiver

    # Подключаемся к SMTP (пример для Gmail)
    with smtplib.SMTP("smtp.gmail.com", 587) as server:
        server.starttls()               # Шифруем соединение
        server.login(sender, password)  # Логинимся
        server.sendmail(sender, [receiver], msg.as_string())

2) Retry-логика для надёжности

utils/retry.py
import time


def with_retry(func, retries: int = 3, delay: float = 1.5):
    last_error = None

    for attempt in range(1, retries + 1):
        try:
            return func()
        except Exception as e:
            last_error = e
            if attempt < retries:
                time.sleep(delay * attempt)  # Увеличиваем паузу

    # Если все попытки неудачны — пробрасываем последнюю ошибку
    raise last_error
