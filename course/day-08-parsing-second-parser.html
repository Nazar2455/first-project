2) Адаптер селекторов под новый сайт

parsers/site2_adapter.py
from bs4 import BeautifulSoup


def parse_site2(html: str) -> list[dict]:
    soup = BeautifulSoup(html, "html.parser")
    cards = soup.select(".listing-item")
    rows = []

    for card in cards:
        title_el = card.select_one(".listing-item__title")
        price_el = card.select_one(".listing-item__price")
        url_el = card.select_one("a.listing-item__link")

        if not title_el or not price_el or not url_el:
            continue

        rows.append({
            "title": title_el.get_text(strip=True),
            "price": price_el.get_text(strip=True),
            "url": url_el.get("href", ""),
        })

    return rows

3) Главный сценарий запуска

main.py (идея)
from services.http_client import fetch_html
from parsers.site2_adapter import parse_site2
from services.export import save_csv, save_json

URL = "https://example-site-2.com/catalog"


def main() -> None:
    html = fetch_html(URL)
    rows = parse_site2(html)

    save_csv(rows, "result_site2.csv")
    save_json(rows, "result_site2.json")

    print(f"Готово: {len(rows)} строк")


if __name__ == "__main__":
    main()
