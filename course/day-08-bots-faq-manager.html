2) Клавиатура FAQ

keyboards/faq_kb.py
from aiogram.types import InlineKeyboardMarkup, InlineKeyboardButton


def get_faq_keyboard() -> InlineKeyboardMarkup:
    return InlineKeyboardMarkup(inline_keyboard=[
        [InlineKeyboardButton(text="💳 Оплата", callback_data="faq:pay")],
        [InlineKeyboardButton(text="🚚 Доставка", callback_data="faq:delivery")],
        [InlineKeyboardButton(text="📦 Гарантия", callback_data="faq:warranty")],
        [InlineKeyboardButton(text="👨‍💼 Связаться с менеджером", callback_data="faq:manager")],
    ])

3) Обработчики callback и контакт менеджера

handlers/faq.py
from aiogram import Router, F
from aiogram.filters import Command
from aiogram.types import Message, CallbackQuery

from keyboards.faq_kb import get_faq_keyboard

router = Router()

FAQ_TEXTS = {
    "pay": "Оплата: карта/перевод, выдаём чек.",
    "delivery": "Доставка: 1-3 дня по РФ.",
    "warranty": "Гарантия: 12 месяцев.",
}

@router.message(Command("start"))
async def start_handler(message: Message):
    await message.answer("Выбери вопрос:", reply_markup=get_faq_keyboard())

@router.callback_query(F.data.startswith("faq:"))
async def faq_callback(callback: CallbackQuery):
    action = callback.data.split(":", 1)[1]

    if action == "manager":
        await callback.message.answer("Менеджер: @your_manager_username")
    else:
        await callback.message.answer(FAQ_TEXTS.get(action, "Вопрос пока не добавлен"))

    await callback.answer()
Этот бот быстрый в продаже: у бизнеса почти всегда есть FAQ и точка контакта менеджера.
