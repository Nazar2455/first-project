1) Мини-теория

  • Фильтры делают данные «клиентскими», а не «сырыми».
  • Мониторинг новых объектов = повторная ценность сервиса.
  • Если ты умеешь показать «что изменилось с прошлого запуска» — это уже полу-аналитический продукт.

2) Полный пример фильтров и экспорта

services/filtering.py
import json
import csv
from pathlib import Path


def parse_price(raw: str) -> int:
    # "12 990 ₽" -> 12990
    digits = ''.join(ch for ch in str(raw) if ch.isdigit())
    return int(digits) if digits else 0


def filter_min_max(rows: list[dict], min_price: int, max_price: int) -> list[dict]:
    result = []
    for row in rows:
        p = parse_price(row.get("price", "0"))
        if min_price <= p <= max_price:
            row["price_num"] = p  # Сохраняем числовую цену
            result.append(row)
    return result


def export_json(rows: list[dict], file_path: str):
    with open(file_path, "w", encoding="utf-8") as f:
        json.dump(rows, f, ensure_ascii=False, indent=2)


def export_csv(rows: list[dict], file_path: str):
    if not rows:
        return
    with open(file_path, "w", encoding="utf-8", newline="") as f:
        writer = csv.DictWriter(f, fieldnames=list(rows[0].keys()))
        writer.writeheader()
        writer.writerows(rows)

3) Мониторинг новых объектов

services/monitoring.py
import json
from pathlib import Path

SNAPSHOT_FILE = Path("last_snapshot.json")


def load_snapshot() -> set[str]:
    # Загружаем прошлый список ID/URL
    if not SNAPSHOT_FILE.exists():
        return set()
    data = json.loads(SNAPSHOT_FILE.read_text(encoding="utf-8"))
    return set(data)


def save_snapshot(ids: set[str]) -> None:
    SNAPSHOT_FILE.write_text(json.dumps(sorted(ids), ensure_ascii=False, indent=2), encoding="utf-8")


def find_new_items(current_rows: list[dict], id_key: str = "url") -> list[dict]:
    old_ids = load_snapshot()

    # Собираем идентификаторы текущего запуска
    current_ids = set(str(r.get(id_key, "")) for r in current_rows if r.get(id_key))

    # Новые — это то, чего не было в прошлом запуске
    new_ids = current_ids - old_ids

    new_rows = [r for r in current_rows if str(r.get(id_key, "")) in new_ids]

    # Обновляем snapshot на следующий запуск
    save_snapshot(current_ids)

    return new_rows
На практике это превращается в услугу «мониторинг новых объявлений/цен».
