1) Сервис экспорта

services/export_service.py
import csv
from db.repository import get_last_leads


async def export_leads_to_csv(file_path: str = "leads_export.csv") -> str:
    # Берём последние заявки из БД
    leads = await get_last_leads(limit=1000)

    # Пишем CSV-файл
    with open(file_path, "w", encoding="utf-8", newline="") as f:
        writer = csv.writer(f)
        writer.writerow(["id", "name", "phone", "comment", "created_at"])  # Заголовки

        for lead in leads:
            lead_id, name, phone, comment, created_at = lead
            writer.writerow([lead_id, name, phone, comment, created_at])

    return file_path

2) Команда /export

handlers/export.py
from aiogram import Router
from aiogram.filters import Command
from aiogram.types import Message, FSInputFile

from utils.security import is_admin
from services.export_service import export_leads_to_csv

router = Router()

@router.message(Command("export"))
async def export_command(message: Message):
    # Проверяем, что пользователь — админ
    if not is_admin(message.from_user.id):
        await message.answer("⛔ Команда только для админа")
        return

    # Создаём CSV
    file_path = await export_leads_to_csv()

    # Отправляем файл в чат
    file = FSInputFile(file_path)
    await message.answer_document(file, caption="📦 Экспорт лидов")
