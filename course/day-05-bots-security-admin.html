1) Мини-теория: зачем ограничения по ADMIN_IDS

  • Команды админа нельзя открывать всем пользователям.
  • Иначе любой сможет читать чужие лиды.
  • Проверка ID — минимальный уровень безопасности в MVP-боте.

2) Маленький пример проверки доступа

utils/security.py (минимальный вариант)
import os
from dotenv import load_dotenv

load_dotenv()

def get_admin_ids() -> set[int]:
    # Берём строку из .env: "123,456,789"
    raw = os.getenv("ADMIN_IDS", "")
    result = set()

    for token in raw.split(","):
        token = token.strip()
        if token.isdigit():
            result.add(int(token))

    return result


def is_admin(user_id: int) -> bool:
    return user_id in get_admin_ids()

3) Полный блок: /admin + последние заявки

handlers/admin.py
from aiogram import Router
from aiogram.filters import Command
from aiogram.types import Message

from utils.security import is_admin
from db.repository import get_last_leads

router = Router()

@router.message(Command("admin"))
async def admin_panel(message: Message):
    user_id = message.from_user.id

    # 1) Проверяем доступ
    if not is_admin(user_id):
        await message.answer("⛔ Доступ запрещён")
        return

    # 2) Достаём последние заявки из БД
    leads = await get_last_leads(limit=10)

    if not leads:
        await message.answer("📭 Пока заявок нет")
        return

    # 3) Формируем удобный текст
    lines = ["📊 Последние заявки:"]
    for lead in leads:
        lead_id, name, phone, comment, created_at = lead
        lines.append(
            f"\n#{lead_id} | {created_at}\n"
            f"Имя: {name}\n"
            f"Телефон: {phone}\n"
            f"Комментарий: {comment}"
        )

    await message.answer("\n".join(lines))
Идея: даже такой простой /admin уже выглядит как полноценный коммерческий функционал.
